     "LoginErrMsg":"登录成功！",
     "Address":"",
     "Birthday":"19841010",
     "Experience":1453333973000,//微妙timestamp
     "Gender":1,
     "PostCode":"111111",
     "VipGrade":30,
     "VipGradeRemark":"钻石贵宾",
     "Email":"wang_peng@163.com",

   }
