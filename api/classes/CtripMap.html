app_locate

(
  • is_async
)

定义在 bridge.js:2969

可用版本 v5.1

定位，定位完成会有2次callback，第一次返回经纬度信息(5.8加入，使用定位缓存的时候，不回掉该结果，直接回掉逆地址解析的数据)，第二次返回逆地址解析的信息

参数:

  • is_async Bool

    true标识是异步定位，false标识为同步定位

举例:

   //调用定位
   CtripUtil.app_locate(true);

   //1. 返回定位的经纬度信息-------5.8版本加入
   var geo_json_obj = 
    {
       tagname:'locate',
       param:{
           "value":{
               lat:'121.487899',
               lng:'31.249162'
           },
           'timeout': '2013/09/12 12:32:36',
       }
   }
   app.callback(geo_json_obj);


   //2. 返回定位的逆地址解析信息
   var json_obj =
   {
       tagname:'locate',
       param:{
           "value":{
               country:"中国",//5.9加入
               countryShortName:"CN",//5.9加入
               city:"上海", //5.9加入
               ctyName: '上海', //后续版本将会废弃，使用city代替
               province: '上海',    //5.8.1版本加入
               district:"浦东新区", //5.8.1版本加入
               addrs:'上海市浦东南路22号',
               lat:'121.487899',
               lng:'31.249162'
           },
           'timeout': '2013/09/12 12:32:36',
           'locateStatus':0,//iOS新增字段:-1网络不通，当前无法定位,-2定位没有开启
       }
   }
   app.callback(json_obj);

app_show_map

(
  • latitude,
  • longitude,
  • title,
  • subtitle,
)

定义在 bridge.js:3035

可用版本 v5.5

在地图上显示某个位置

参数:

  • latitude, Double

    纬度

  • longitude, Double

    经度

  • title, String

    在地图上显示的点的主标题

  • subtitle, String

    在地图上显示点的附标题

举例:

   CtripMap.app_show_map(31.3222323, 121.32232332, "上海野生动物园", "浦东新区陆家嘴1234号");

app_show_map_with_POI_list

(
  • poiList
)

定义在 bridge.js:3081

可用版本 v5.8

在地图上显示多个POI位置点

参数:

  • poiList Array

    poi列表, list中为JSON对象，key为：latitude, longitude, title, subtitle

举例:

   var poi0 = {};
   poi0.latitude = 31.3222323; //must
   poi0.longitude = 121.32232332;//must
   poi0.title = "上海野生动物园"; //must
   poi0.subtitle = "浦东新区陆家嘴1234号";//optional

   var poi1 = {};
   poi1.latitude = 30.3222323; //must
   poi1.longitude = 120.32232332;//must
   poi1.title = "上海野生动物园A"; //must
   poi1.subtitle = "浦东新区陆家嘴1234号A";//optional


   var poi2 = {};
   poi2.latitude = 32.3222323; //must
   poi2.longitude = 122.32232332;//must
   poi2.title = "上海野生动物园B"; //must
   poi2.subtitle = "浦东新区陆家嘴1234号B";//optional

   var poiList = new Array();
   poiList[0] = poi0;
   poiList[1] = poi1;
   poiList[2] = poi2;

   CtripMap.app_show_map(poiList);
