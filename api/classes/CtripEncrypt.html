app_ctrip_encrypt

(
  • inString
  • encType
)

定义在 bridge.js:1966

可用版本 5.5

Ctrip私有加解密算法

参数:

  • inString String

    需要做加解密的字符串

  • encType String

    加解密类型，加密为1， 解密为2，其它不处理

举例:

 CtripEncrypt.app_ctrip_encrypt("abcdxxxx",1);
 //调用后，H5会收到native回调的数据
   var json_obj =
   {
       tagname:"ctrip_encrypt",
       param:
       {
           inString:"abcdxxxx",
           outString:"ABScdXkYZunwXVF5kQpffnY+oL/MFmJGkn8ra8Ab5cI=",
           encType:1
       },
   }
   app.callback(json_obj);
